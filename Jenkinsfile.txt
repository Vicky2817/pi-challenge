pipeline {
    agent any
    stages {
        stage('Clone Repository') {
            steps {
                script {
                    // Ensure repository is pulled correctly
                    checkout scm
                }
            }
        }
        stage('List Files') {
            steps {
                // Check if the script is available in the workspace
                sh 'ls -lah'
            }
        }
        stage('Run Algorithm') {
            steps {
                script {
                    // Make sure the script is in the correct path and executable
                    sh 'chmod +x ./algorithm.sh'
                    sh './algorithm.sh'
                }
            }
        }
    }
}

pipeline {
    agent any
    stages {
        stage('Clone Repository') {
            steps {
                script {
                    checkout scm
                }
            }
        }
        stage('Debug File') {
            steps {
                sh 'ls -lah'   // List files
                sh 'file algorithm.sh'  // Check file type
                sh 'cat -A algorithm.sh'  // Show special characters
            }
        }
        stage('Run Algorithm') {
            steps {
                script {
                    // Convert CRLF to LF
                    sh 'apt-get update && apt-get install -y dos2unix || yum install -y dos2unix'
                    sh 'dos2unix algorithm.sh'
        
                    // Ensure script is executable
                    sh 'chmod +x algorithm.sh'
                    sh './algorithm.sh'
                }
            }
        }
    }
}
